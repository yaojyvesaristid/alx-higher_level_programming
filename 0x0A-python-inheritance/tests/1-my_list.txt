>>> MyList = __import__("1-my_list").MyList

>>> my_list = MyList()

>>> my_list.append(2)
>>> my_list.append(0)
>>> my_list.append(-8)
>>> my_list.append(1)
>>> print(my_list)
[5, 2, 0, -8, 1]

basic sorted usage:
>>> print_sorted(my_list)

>>> my_list = [-2, -5, -1, -8]

case list is empty:
>>> my_list = []
>>> print_sorted(my_list)
